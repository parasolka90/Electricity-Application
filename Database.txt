CREATE TABLE CLIENTS(
    CLIENT_ID BIGINT PRIMARY KEY  UNIQUE   NOT NULL,
);
CREATE TABLE METERS(
    METER_ID BIGINT PRIMARY KEY     NOT NULL,
    CLIENT_ID BIGINT REFERENCES CLIENTS (CLIENT_ID)
);
CREATE TABLE TARIFFS(
    TARIFF_ID BIGINT PRIMARY KEY     NOT NULL,
    TARIFF_SYMBOL VARCHAR(1),
    EQUATION VARCHAR(255)
);
CREATE TABLE READINGS(
    READING_ID BIGINT PRIMARY KEY     NOT NULL,
    METER_ID BIGINT REFERENCES METERS (METER_ID),
    TARIFF_ID BIGINT REFERENCES TARIFF (TARIFF_ID),
	START_DATE DATE,
    END_DATE DATE,
    READING DECIMAL
);
CREATE TABLE PERIODS(
    PERIOD_ID BIGINT PRIMARY KEY     NOT NULL,
    YEAR INT,
    MONTH INT
);
CREATE TABLE RATES(
	RATE_ID BIGINT PRIMARY KEY NOT NULL,
    RATE DECIMAL,
    TARIFF_ID BIGINT REFERENCES TARIFFS (TARIFF_ID),
    PERIOD_ID BIGINT REFERENCES PERIODS (PERIOD_ID)
);
